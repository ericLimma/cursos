            let carros = [
            'Fusca',
            'Gol',
            'Civic',
            'Corolla',
            'Onix',
            'Renegade',
            'Hilux',
            'C4',
            'Argo',
            'Fusion'
          ];

          let display1 = document.getElementById('display-1');
          let displayList = '';

          for (let i = 0; i < carros.length; i++) {
            displayList += carros[i];

              let x = 1;
              let result = '';
              let display2 = document.getElementById('display-2')

                let y = 0;
