Java Script - Estruturas de Repetição

While(condicao)


          while(x <= 10){ //condicao de saída.
            document.write(x);
            x++ //instrução de incremento;
          }; // tradução: faça enquando x for menor que ou igual a 10.
          for (let i = 0; i <= 10; i++) { // o incremento é feito aqui.
            escrever(i);
          };

For


        for (let i = 0; i <= 10; i++) {
          escrever(i);
          if (i == 5) {
            break; //interrompe o laço
          }
        };
        for (let i = 0; i <= 10; i++) {
          escrever(i);
          if (i == 5) {
            continue; //interrompe o laço
          }
        };

Set Interval


        let y = 0;

        let display = document.getElementById('display');

        const intervalId = setInterval(() => {

          display.textContent += `${y}, `;

          y++;

          if (y > 10) {
            clearInterval(intervalId);
            display.textContent += 'Acabou';
          }
        }, 200);

Array Encadeado


        let display5 = document.getElementById('display-5');

        for (let i = 0; i <= 10; i++) {
          for (let m = 0; m <= 10; m++) {
            let resultado = `${i} x ${m} = ${i * m} <br>`;
  
            display5.innerHTML += `${resultado}`
          }
          display5.innerHTML += `<hr>`
        } // faz a multiplicação de um array com a outra.
